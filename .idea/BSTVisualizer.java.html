import javax.swing.*;
import java.awt.*;

public class BSTVisualizer extends JPanel {

    BinarySearchTree bst;

    public BSTVisualizer(BinarySearchTree bst) {
        this.bst = bst;
    }

    protected void paintComponent(Graphics g) {
        super.paintComponent(g);
        drawTree(g, getWidth() / 2, 30, bst.root, getWidth() / 4);
    }

    private void drawTree(Graphics g, int x, int y, Node node, int xOffset) {

        if (node == null) {
            return;
        }

        // Draw node (oval)
        g.setColor(Color.BLACK);
        int nodeRadius = 30;
        int ovalX = x - nodeRadius / 2;
        int ovalY = y - nodeRadius / 2;
        g.drawOval(ovalX, ovalY, nodeRadius, nodeRadius);

        // Draw centered text in the oval
        FontMetrics fm = g.getFontMetrics();
        String nodeValue = Integer.toString(node.data);
        int textWidth = fm.stringWidth(nodeValue);
        int textHeight = fm.getAscent();
        g.drawString(nodeValue, x - textWidth / 2, y + textHeight / 4);

        // Draw lines to children
        int lineY = y + nodeRadius / 2; // Start of line from the bottom of the oval

        if (node.left != null) {
            int childX = x - xOffset;
            int childY = y + 60; // Adjust distance between levels
            g.setColor(Color.BLACK);
            g.drawLine(x, lineY, childX, childY - nodeRadius / 2); // Line connects to the top of the child oval
            drawTree(g, childX, childY, node.left, xOffset / 2);
        }

        if (node.right != null) {
            int childX = x + xOffset;
            int childY = y + 60; // Adjust distance between levels
            g.setColor(Color.BLACK);
            g.drawLine(x, lineY, childX, childY - nodeRadius / 2); // Line connects to the top of the child oval
            drawTree(g, childX, childY, node.right, xOffset / 2);
        }
    }
}
